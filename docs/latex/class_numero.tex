\doxysubsection{Detailed Description}
Representa o número de um quarto. 

A classe \doxylink{class_numero}{Numero} armazena e valida números de quartos dentro do intervalo permitido pelo sistema.

{\bfseries{FORMATO VÁ\+LIDO\+:}}
\begin{DoxyItemize}
\item Valores de 001 a 999
\end{DoxyItemize}

{\bfseries{REGRAS DE VALIDAÇÃO\+:}}
\begin{DoxyItemize}
\item Valor mínimo\+: 1 (representado como 001)
\item Valor máximo\+: 999
\item Armazenado como inteiro (1 a 999)
\item Exibido com 3 dígitos (001, 002, ..., 999)
\end{DoxyItemize}

{\bfseries{EXEMPLOS\+:}}
\begin{DoxyItemize}
\item ✅ Válido\+: 1 (exibido como 001)
\item ✅ Válido\+: 101 (quarto 101)
\item ✅ Válido\+: 999 (último quarto permitido)
\item ❌ Inválido\+: 0 (abaixo do mínimo)
\item ❌ Inválido\+: 1000 (acima do máximo)
\item ❌ Inválido\+: -\/5 (valor negativo)
\end{DoxyItemize}

{\bfseries{NOTA DE USO\+:}}
\begin{DoxyItemize}
\item Cada quarto deve ter número único no sistema
\item Números não utilizados podem ser reutilizados após exclusão
\item O número é a chave primária do quarto 
\end{DoxyItemize}
