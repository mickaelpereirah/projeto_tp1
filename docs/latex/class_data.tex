\doxysubsection{Detailed Description}
Representa uma data válida no sistema. 

A classe \doxylink{class_data}{Data} armazena e valida datas no formato dia-\/mês-\/ano, com validação completa de dias válidos por mês, incluindo tratamento de anos bissextos.

{\bfseries{FORMATO VÁ\+LIDO\+:}}
\begin{DoxyItemize}
\item dia-\/mês-\/ano
\item Dia é número de 1 a 31
\item Mês é JAN, FEV, MAR, ABR, MAI, JUN, JUL, AGO, SET, OUT, NOV ou DEZ
\item Ano é número de 2000 a 2999
\item \doxylink{class_data}{Data} deve ser válida considerando os anos bissextos
\end{DoxyItemize}

{\bfseries{REGRAS DE VALIDAÇÃO\+:}}
\begin{DoxyEnumerate}
\item {\bfseries{Formato}}\+: DD-\/\+MMM-\/\+AAAA onde\+:
\begin{DoxyItemize}
\item DD = dia (01 a 31)
\item MMM = mês por extenso em maiúsculas (JAN, FEV, MAR, ABR, MAI, JUN, JUL, AGO, SET, OUT, NOV, DEZ)
\item AAAA = ano (2000 a 2999)
\end{DoxyItemize}
\item {\bfseries{Dia}}\+: Deve ser válido para o mês\+:
\begin{DoxyItemize}
\item Meses com 31 dias\+: JAN, MAR, MAI, JUL, AGO, OUT, DEZ
\item Meses com 30 dias\+: ABR, JUN, SET, NOV
\item FEV\+: 28 dias (anos normais) ou 29 dias (anos bissextos)
\end{DoxyItemize}
\item {\bfseries{Ano Bissexto}}\+: Um ano é bissexto se\+:
\begin{DoxyItemize}
\item É divisível por 4 E não é divisível por 100, OU
\item É divisível por 400
\item Exemplos\+: 2000 (bissexto), 2004 (bissexto), 2100 (não bissexto)
\end{DoxyItemize}
\end{DoxyEnumerate}

{\bfseries{EXEMPLOS\+:}}
\begin{DoxyItemize}
\item ✅ Válido\+: "{}15-\/\+JAN-\/2024"{} (15 de janeiro de 2024)
\item ✅ Válido\+: "{}29-\/\+FEV-\/2024"{} (2024 é bissexto)
\item ✅ Válido\+: "{}31-\/\+DEZ-\/2999"{} (último dia do intervalo)
\item ✅ Válido\+: "{}01-\/\+JAN-\/2000"{} (primeiro dia do intervalo)
\item ❌ Inválido\+: "{}29-\/\+FEV-\/2023"{} (2023 não é bissexto)
\item ❌ Inválido\+: "{}31-\/\+ABR-\/2024"{} (abril tem apenas 30 dias)
\item ❌ Inválido\+: "{}32-\/\+JAN-\/2024"{} (dia inválido)
\item ❌ Inválido\+: "{}15-\/13-\/2024"{} (mês deve ser por extenso)
\item ❌ Inválido\+: "{}15-\/jan-\/2024"{} (mês deve estar em maiúsculas)
\item ❌ Inválido\+: "{}15/\+JAN/2024"{} (separador deve ser hífen)
\item ❌ Inválido\+: "{}15-\/\+JAN-\/1999"{} (ano antes de 2000)
\item ❌ Inválido\+: "{}15-\/\+JAN-\/3000"{} (ano depois de 2999)
\end{DoxyItemize}
