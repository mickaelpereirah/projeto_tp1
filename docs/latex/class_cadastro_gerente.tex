\doxysubsection{Detailed Description}
Serviço responsável pelo cadastro de gerentes no sistema. 

Esta classe gerencia o processo completo de cadastro de um novo gerente, incluindo coleta de dados, validação e armazenamento no container.

{\bfseries{DADOS COLETADOS\+:}}
\begin{DoxyItemize}
\item \doxylink{class_nome}{Nome} (5-\/20 caracteres, capitalizado)
\item \doxylink{class_email}{Email} (único no sistema, usado como chave)
\item \doxylink{class_ramal}{Ramal} (00-\/50)
\item \doxylink{class_senha}{Senha} (5 caracteres com requisitos de segurança)
\end{DoxyItemize}

{\bfseries{VALIDAÇÕ\+ES\+:}}
\begin{DoxyItemize}
\item \doxylink{class_email}{Email} não pode estar duplicado no sistema
\item Todos os domínios são validados individualmente
\item Dados são validados antes de criar o objeto \doxylink{class_gerente}{Gerente}
\end{DoxyItemize}

{\bfseries{FLUXO DE CADASTRO\+:}}
\begin{DoxyEnumerate}
\item Solicita dados via métodos herdados de \doxylink{class_cadastro}{Cadastro}
\item Cria objeto \doxylink{class_gerente}{Gerente} com dados validados
\item Verifica se email já existe no container
\item Inclui no \doxylink{class_container_gerente}{Container\+Gerente} via \doxylink{class_central_containers}{Central\+Containers}
\item Retorna sucesso ou falha 
\end{DoxyEnumerate}
