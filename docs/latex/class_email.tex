\doxysubsection{Detailed Description}
Representa um endereço de email válido. 

A classe \doxylink{class_email}{Email} armazena e valida endereços de email seguindo regras específicas para parte local e domínio.

{\bfseries{FORMATO VÁ\+LIDO\+:}}
\begin{DoxyItemize}
\item parte-\/local@domínio
\end{DoxyItemize}

{\bfseries{REGRAS DA PARTE LOCAL (antes do @)\+:}}
\begin{DoxyItemize}
\item Pode conter\+: letra (a-\/z), dígito (0-\/9), ponto (.) ou hífen (-\/)
\item NÃO pode iniciar ou terminar com ponto ou hífen
\item Ponto ou hífen deve ser seguido por letra(s) ou dígito(s)
\item Comprimento máximo\+: 64 caracteres
\end{DoxyItemize}

{\bfseries{REGRAS DO DOMÍ\+NIO (após o @)\+:}}
\begin{DoxyItemize}
\item Composto por uma ou mais partes separadas por ponto (.)
\item Cada parte pode conter\+: letra (a-\/z), dígito (0-\/9) ou hífen (-\/)
\item NÃO pode iniciar ou terminar com hífen
\item Comprimento máximo\+: 255 caracteres
\end{DoxyItemize}

{\bfseries{EXEMPLOS\+:}}
\begin{DoxyItemize}
\item ✅ Válido\+: "{}usuario@exemplo.\+com"{}
\item ✅ Válido\+: "{}nome.\+sobrenome@empresa.\+com.\+br"{}
\item ✅ Válido\+: "{}user-\/name@dominio.\+org"{}
\item ✅ Válido\+: "{}contato123@site.\+com"{}
\item ✅ Válido\+: "{}a@b.\+co"{} (mínimo)
\item ❌ Inválido\+: "{}.\+usuario@exemplo.\+com"{} (começa com ponto)
\item ❌ Inválido\+: "{}usuario.@exemplo.\+com"{} (termina com ponto)
\item ❌ Inválido\+: "{}usuario@"{} (falta domínio)
\item ❌ Inválido\+: "{}@exemplo.\+com"{} (falta parte local)
\item ❌ Inválido\+: "{}usuario..\+nome@exemplo.\+com"{} (pontos consecutivos)
\item ❌ Inválido\+: "{}usuario@-\/exemplo.\+com"{} (domínio começa com hífen)
\item ❌ Inválido\+: "{}usuario@exemplo-\/.\+com"{} (domínio termina com hífen)
\item ❌ Inválido\+: "{}usuario nome@exemplo.\+com"{} (espaço não permitido)
\end{DoxyItemize}

{\bfseries{DETALHES TÉ\+CNICOS\+:}}
\begin{DoxyItemize}
\item Total máximo\+: 64 (local) + 1 (@) + 255 (domínio) = 320 caracteres
\item Apenas caracteres ASCII básicos (sem acentuação)
\item Case-\/insensitive (\href{mailto:user@EXEMPLO.com}{\texttt{user@\+EXEMPLO.\+com}} = \href{mailto:user@exemplo.com}{\texttt{user@exemplo.\+com}}) 
\end{DoxyItemize}
