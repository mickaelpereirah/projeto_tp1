\doxysubsection{Detailed Description}
Representa um número de telefone internacional. 

A classe \doxylink{class_telefone}{Telefone} armazena e valida números telefônicos no formato internacional, começando com símbolo de adição (+) seguido por dígitos.

{\bfseries{FORMATO VÁ\+LIDO\+:}}
\begin{DoxyItemize}
\item +DDDDDDDDDDDDDD
\item D é dígito (0-\/9)
\end{DoxyItemize}

{\bfseries{REGRAS DE VALIDAÇÃO\+:}}
\begin{DoxyItemize}
\item Deve começar com o símbolo + (mais)
\item Seguido por exatamente 14 dígitos (0-\/9)
\item Não são permitidos\+:
\begin{DoxyItemize}
\item Espaços em branco
\item Parênteses
\item Hífens ou outros separadores
\item Letras ou caracteres especiais (exceto o + inicial)
\end{DoxyItemize}
\item Total\+: 15 caracteres (1 símbolo + e 14 dígitos)
\end{DoxyItemize}

{\bfseries{ESTRUTURA TÍ\+PICA\+:}}
\begin{DoxyItemize}
\item Código do país\+: 1-\/3 dígitos (ex\+: +55 para Brasil)
\item Código de área\+: 2 dígitos (ex\+: 61 para Brasília)
\item Número\+: restante dos dígitos
\end{DoxyItemize}

{\bfseries{EXEMPLOS\+:}}
\begin{DoxyItemize}
\item ✅ Válido\+: "{}+55619876543210"{} (Brasil, Brasília)
\item ✅ Válido\+: "{}+12025551234000"{} (EUA)
\item ✅ Válido\+: "{}+44207946000000"{} (Reino Unido)
\item ✅ Válido\+: "{}+00000000000000"{} (número de teste)
\item ❌ Inválido\+: "{}55619876543210"{} (falta o símbolo +)
\item ❌ Inválido\+: "{}+5561987654321"{} (apenas 13 dígitos, precisa de 14)
\item ❌ Inválido\+: "{}+556198765432100"{} (15 dígitos, precisa de 14)
\item ❌ Inválido\+: "{}+55 61 98765 4321"{} (contém espaços)
\item ❌ Inválido\+: "{}+55-\/61-\/98765-\/4321"{} (contém hífens)
\item ❌ Inválido\+: "{}+55(61)987654321"{} (contém parênteses)
\item ❌ Inválido\+: "{}55-\/61-\/98765-\/4321"{} (falta + e tem hífens)
\item ❌ Inválido\+: "{}+5561987654321a"{} (contém letra)
\end{DoxyItemize}
